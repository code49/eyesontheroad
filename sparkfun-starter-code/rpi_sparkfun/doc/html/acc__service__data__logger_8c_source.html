188 int main(int argc, char *argv[])
189 {
190  input_t input;
191 
192  initialize_input(&input);
193 
194  signal(SIGINT, interrupt_handler);
195 
196  if (!parse_options(argc, argv, &input))
197  {
198  if (input.file_path != NULL)
199  {
200  acc_integration_mem_free(input.file_path);
201  }
202 
203  return EXIT_FAILURE;
204  }
205 
206  acc_hal_t hal = *acc_hal_integration_get_implementation();
207 
208  hal.log.log_level = input.log_level;
209 
210  if (!acc_rss_activate(&hal))
211  {
212  return EXIT_FAILURE;
213  }
214 
215  FILE *file = stdout;
216 
217  if (input.file_path != NULL)
218  {
219  file = fopen(input.file_path, "w");
220 
221  if (file == NULL)
222  {
223  acc_integration_mem_free(input.file_path);
224  printf("opening output file failed\n");
225  return EXIT_FAILURE;
226  }
227  }
228 
229  bool service_status = false;
230 
231  switch (input.service_type)
232  {
233  case POWER_BIN:
234  {
235  acc_service_configuration_t power_bin_configuration = set_up_power_bin(&input);
236 
237  if (power_bin_configuration != NULL)
238  {
239  service_status = execute_power_bin(power_bin_configuration, file, input.wait_for_interrupt,
240  input.update_count, input.metadata_options);
241 
242  if (!service_status)
243  {
244  printf("execute_power_bin() failed\n");
245  }
246  }
247 
248  acc_service_power_bins_configuration_destroy(&power_bin_configuration);
249  break;
250  }
251 
252  case ENVELOPE:
253  {
254  acc_service_configuration_t envelope_configuration = set_up_envelope(&input);
255 
256  if (envelope_configuration != NULL)
257  {
258  service_status = execute_envelope(envelope_configuration, file, input.wait_for_interrupt,
259  input.update_count, input.metadata_options);
260 
261  if (!service_status)
262  {
263  printf("execute_envelope() failed\n");
264  }
265  }
266 
267  acc_service_envelope_configuration_destroy(&envelope_configuration);
268  break;
269  }
270 
271  case IQ:
272  {
273  acc_service_configuration_t iq_configuration = set_up_iq(&input);
274 
275  if (iq_configuration != NULL)
276  {
277  service_status = execute_iq(iq_configuration, file, input.wait_for_interrupt,
278  input.update_count, input.metadata_options);
279 
280  if (!service_status)
281  {
282  printf("execute_iq() failed\n");
283  }
284  }
285 
286  acc_service_iq_configuration_destroy(&iq_configuration);
287  break;
288  }
289 
290  case SPARSE:
291  {
292  acc_service_configuration_t sparse_configuration = set_up_sparse(&input);
293 
294  if (sparse_configuration != NULL)
295  {
296  service_status = execute_sparse(sparse_configuration, file, input.wait_for_interrupt,
297  input.update_count, input.metadata_options, input.sparse_data_format);
298 
299  if (!service_status)
300  {
301  printf("execute_sparse() failed\n");
302  }
303  }
304 
305  acc_service_sparse_configuration_destroy(&sparse_configuration);
306  break;
307  }
308 
309  default:
310  {
311  printf("Invalid service_type %d\n", input.service_type);
312  }
313  }
314 
315  if (input.file_path != NULL)
316  {
317  acc_integration_mem_free(input.file_path);
318  fclose(file);
319  }
320 
321  acc_rss_deactivate();
322 
323  return service_status ? EXIT_SUCCESS : EXIT_FAILURE;
324 }

383 static bool string_to_power_save_mode(const char *str, acc_power_save_mode_t *power_save_mode)
384 {
385  if (strcmp("ACTIVE", str) == 0)
386  {

390  {

398  {
399  *power_save_mode = ACC_POWER_SAVE_MODE_HIBERNATE;
400  }
401  else if (strcmp("OFF", str) == 0)
402  {
403  *power_save_mode = ACC_POWER_SAVE_MODE_OFF;
404  }
405  else
406  {
407  return false;
408  }
409 
410  return true;
411 }

414 static bool parse_options(int argc, char *argv[], input_t *input)
415 {
416  static struct option long_options[] =
417  {
418  {"service-type", required_argument, 0, 't'},
419  {"sweep-count", required_argument, 0, 'c'},
420  {"range-start", required_argument, 0, 'b'},
421  {"range-end", required_argument, 0, 'e'},
422  {"frequency", required_argument, 0, 'f'},
423  {"power-save-mode", required_argument, 0, 'p'},
424  {"gain", required_argument, 0, 'g'},
425  {"downsampling-factor", required_argument, 0, 'd'},
426  {"hwaas", required_argument, 0, 'a'},
427  {"number-of-bins", required_argument, 0, 'n'},
428  {"sweeps-per-frame", required_argument, 0, 'm'},
429  {"sparse-data-format", required_argument, 0, 'k'},
430  {"out", required_argument, 0, 'o'},
431  {"service-profile", required_argument, 0, 'y'},
432  {"running-avg-factor", required_argument, 0, 'r'},
433  {"integer-iq", no_argument, 0, 'i'},
434  {"sensor", required_argument, 0, 's'},
435  {"runtime", no_argument, 0, 'u'},
436  {"date-timestamp", no_argument, 0, 'U'},
437  {"data-warnings", no_argument, 0, 'w'},
438  {"verbose", no_argument, 0, 'v'},
439  {"help", no_argument, 0, 'h'},
440  {NULL, 0, NULL, 0}
441  };
442 
443  int16_t character_code;
444  int32_t option_index = 0;
445 
446  while ((character_code = getopt_long(argc, argv, "t:c:b:e:f:p:g:d:a:n:m:k:o:r:is:uUwvh?:y:", long_options, &option_index)) != -1)
447  {
448  switch (character_code)
449  {
450  case 't':
451  {
452  switch (atoi(optarg))
453  {
454  case 0:
455  {
456  input->service_type = POWER_BIN;
457  break;
458  }
459  case 1:
460  {
461  input->service_type = ENVELOPE;
462  break;
463  }
464  case 2:
465  {
466  input->service_type = IQ;
467  break;
468  }
469  case 3:
470  {
471  input->service_type = SPARSE;
472  break;
473  }
474  default:
475  printf("Invalid service type.\n");
476  print_usage();
477  return false;
478  }
479 
480  break;
481  }
482  case 'c':
483  {
484  input->update_count = atoi(optarg);
485  input->wait_for_interrupt = false;
486  break;
487  }
488  case 'b':
489  {
490  input->start_m = strtof(optarg, NULL);
491  break;
492  }
493  case 'e':
494  {
495  input->end_m = strtof(optarg, NULL);
496  break;
497  }
498  case 'f':
499  {
500  float f = strtof(optarg, NULL);
501  if (f > 0 && f < 100000)
502  {
503  input->frequency = f;
504  }
505  else if (strcmp(optarg, "max") == 0)
506  {
507  input->frequency = INFINITY;
508  }
509  else
510  {
511  printf("Frequency out of range.\n");
512  print_usage();
513  exit(EXIT_FAILURE);
514  }
515 
516  break;
517  }
518  case 'p':
519  {
520  if (!string_to_power_save_mode(optarg, &input->power_save_mode))
521  {
522  printf("Invalid power save mode: %s\n", optarg);
523  print_usage();
524  exit(EXIT_FAILURE);
525  }
526 
527  break;
528  }
529  case 'g':
530  {
531  float g = strtof(optarg, NULL);
532  if (g >= 0 && g <= 1)
533  {
534  input->gain = g;
535  }
536  else
537  {
538  printf("Gain out of range.\n");
539  print_usage();
540  exit(EXIT_FAILURE);
541  }
542 
543  break;
544  }
545  case 'd':
546  {
547  int d = atoi(optarg);
548  if (d > 0 && d < 200)
549  {
550  input->downsampling_factor = d;
551  }
552  else
553  {
554  printf("Downsampling factor out of range.\n");
555  print_usage();
556  exit(EXIT_FAILURE);
557  }
558 
559  break;
560  }
561  case 'a':
562  {
563  int a = atoi(optarg);
564  if (a > 0 && a < 64)
565  {
566  input->hwaas = a;
567  }
568  else
569  {
570  printf("Hardware accelerated average samples out of range.\n");
571  print_usage();
572  exit(EXIT_FAILURE);
573  }
574 
575  break;
576  }
577  case 'n':
578  {
579  int n = atoi(optarg);
580  if (n > 0 && n <= 32)
581  {
582  input->n_bins = n;
583  }
584  else
585  {
586  printf("Number of bins out of range.\n");
587  print_usage();
588  exit(EXIT_FAILURE);
589  }
590 
591  break;
592  }
593  case 'm':
594  {
595  int n = atoi(optarg);
596  if (n > 0 && n < 2048)
597  {
598  input->sweeps_per_frame = n;
599  }
600  else
601  {
602  printf("Number of sweeps per frame out of range.\n");
603  print_usage();
604  exit(EXIT_FAILURE);
605  }
606 
607  break;
608  }
609  case 'k':
610  {
611  int i;
612  for (i = 0; optarg[i] != '\0'; i++)
613  {
614  char c = optarg[i];
615  if (c != 'a' && c != 'c' && c != 'd' && c != 'f')
616  {
617  printf("Bad character \"%c\" in sparse format string.\n", c);
618  print_usage();
619  exit(EXIT_FAILURE);
620  }
621 
622  if (i >= SPARSE_DATA_FORMAT_BUFSIZE - 1)
623  {
624  printf("Too long sparse format string.\n");
625  print_usage();
626  exit(EXIT_FAILURE);
627  }
628 
629  input->sparse_data_format[i] = c;
630  }
631 
632  input->sparse_data_format[i] = '\0';
633 
634  break;
635  }
636  case 'y':
637  {
638  input->service_profile = atoi(optarg);
639  break;
640  }
641  case 'o':
642  {
643  input->file_path = acc_integration_mem_alloc(sizeof(char) * (strlen(optarg) + 1));
644  if (input->file_path == NULL)
645  {
646  printf("Failed allocating memory\n");
647  return false;
648  }
649 
650  snprintf(input->file_path, strlen(optarg) + 1, "%s", optarg);
651  break;
652  }
653  case 'r':
654  {
655  float r = strtof(optarg, NULL);
656  if (r >= 0 && r <= 1)
657  {
658  input->running_avg = r;
659  }
660  else
661  {
662  printf("Running average factor out of range.\n");
663  print_usage();
664  exit(EXIT_FAILURE);
665  }
666 
667  break;
668  }
669  case 'i':
670  {
671  input->integer_iq = true;
672  break;
673  }
674  case 's':
675  {
676  int s = atoi(optarg);
677  if (s > 0 && s <= 4)
678  {
679  input->sensor = s;
680  }
681  else
682  {
683  printf("Sensor id out of range.\n");
684  print_usage();
685  exit(EXIT_FAILURE);
686  }
687 
688  break;
689  }
690  case 'u':
691  {
692  input->metadata_options.runtime = true;
693  break;
694  }
695  case 'U':
696  {
697  input->metadata_options.date_timestamp = true;
698  break;
699  }
700  case 'w':
701  {
702  input->metadata_options.data_warnings = true;
703  break;
704  }
705  case 'v':
706  {
707  input->log_level = ACC_LOG_LEVEL_VERBOSE;
708  break;
709  }
710  case 'h':
711  case '?':
712  {
713  print_usage();
714  return false;
715  }
716  }
717  }
718 
719  if (input->service_type == INVALID_SERVICE)
720  {
721  printf("Missing option service type.\n");
722  print_usage();
723  return false;
724  }
725 
726  return true;
727 }

1247 static void print_sparse_data_item(FILE *file, const uint16_t *data, uint16_t sweep_length, uint16_t sweep_count,
1248  char item_selection)
1249 {
1250  uint16_t data_length = sweep_length * sweep_count;
1251 
1252  switch (item_selection)
1253  {
1254  case 'a':
1255  {
1256  // Average over sweeps
1257  for (uint16_t index = 0; index < sweep_length; index++)
1258  {
1259  int32_t sum = 0;
1260  for (uint16_t k = 0; k < sweep_count; k++)
1261  {
1262  sum += data[k * sweep_length + index];
1263  }
1264 
1265  fprintf(file, "%u\t", (unsigned int)(sum / sweep_count));
1266  }
1267 
1268  break;
1269  }
1270  case 'c':
1271  {
1272  // Average absolute difference between consecutive sweeps
1273  for (uint16_t index = 0; index < sweep_length; index++)
1274  {
1275  int32_t sum = 0;
1276  for (uint16_t k = 1; k < sweep_count; k++)
1277  {
1278  uint16_t i = k * sweep_length + index;
1279  sum += abs(data[i] - data[i - sweep_length]);
1280  }
1281 
1282  unsigned int result = sweep_count > 1 ? sum / (sweep_count - 1) : 0;
1283  fprintf(file, "%u\t", result);
1284  }
1285 
1286  break;
1287  }
1288  case 'd':
1289  {
1290  // Average absolute deviation from the average overs sweeps
1291  for (uint16_t index = 0; index < sweep_length; index++)
1292  {
1293  int32_t sum = 0;
1294  for (uint16_t k = 0; k < sweep_count; k++)
1295  {
1296  sum += data[k * sweep_length + index];
1297  }
1298 
1299  int32_t average = sum / sweep_count;
1300  sum = 0;
1301  for (uint16_t k = 0; k < sweep_count; k++)
1302  {
1303  sum += abs(data[k * sweep_length + index] - average);
1304  }
1305 
1306  fprintf(file, "%u\t", (unsigned int)(sum / sweep_count));
1307  }
1308 
1309  break;
1310  }
1311  case 'f':
1312  {
1313  // Full data frame
1314  for (uint16_t index = 0; index < data_length; index++)
1315  {
1316  fprintf(file, "%u\t", (unsigned int)data[index]);
1317  }
