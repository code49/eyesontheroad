25 static bool gpio_open(int pin, gpio_direction_t direction)
26 {
27  gpios[pin].line = gpiod_chip_get_line(chip, pin);
28  if (gpios[pin].line == NULL)
29  {
30  perror("Failed opening line");
31  return false;
32  }
33 
34  gpios[pin].direction = direction;
35 
36  int res;
37 
38  switch (direction)
39  {
40  case GPIO_DIR_INPUT_INTERRUPT:
41  res = gpiod_line_request_rising_edge_events(gpios[pin].line,
42  GPIOD_CONSUMER);
43  break;
44  case GPIO_DIR_OUTPUT_LOW:
45  case GPIO_DIR_OUTPUT_HIGH:
46  res = gpiod_line_request_output(gpios[pin].line,
47  GPIOD_CONSUMER,
48  direction == GPIO_DIR_OUTPUT_HIGH ? PIN_HIGH : PIN_LOW);
49  break;
50  default:
51  res = -1;
52  assert(false);
53  }
54 
55  if (res != 0)
56  {
57  perror("gpiod_line_request_ failed");
58  return false;
59  }
60 
61  return true;
62 }

65 bool acc_libgpiod_init(const gpio_config_t *pin_config)
66 {
67  int pin = 0;
68 
69  for (pin = 0; pin < GPIO_PIN_COUNT; pin++)
70  {
71  gpios[pin].direction = GPIO_DIR_UNKNOWN;
72  }
73 
74  chip = gpiod_chip_open_by_name(RPI_GPIO_CHIPNAME);
75  if (chip == NULL)
76  {
77  perror("gpiod_chip_open_by_name failed");
78  return false;
79  }
80 
81  pin = 0;
82  while (pin_config[pin].direction != GPIO_DIR_UNKNOWN)
83  {
84  if (!gpio_open(pin_config[pin].pin, pin_config[pin].direction))
85  {
86  return false;
87  }
88 
89  pin++;
90  }
91  return true;
92 }

95 void acc_libgpiod_deinit(void)
96 {
97  if (chip != NULL)
98  {
99  for (int pin = 0; pin < GPIO_PIN_COUNT; pin++)
100  {
101  if (gpios[pin].direction != GPIO_DIR_UNKNOWN)
102  {
103  gpiod_line_release(gpios[pin].line);
104  gpios[pin].direction = GPIO_DIR_UNKNOWN;
105  }
106  }
107 
108  gpiod_chip_close(chip);
109  chip = NULL;
110  }
111 }

130 static unsigned int get_elapsed_ms(struct timespec *start)
131 {
132  struct timespec now;
133 
134  clock_gettime(CLOCK_MONOTONIC, &now);
135  int elapsed = (((now.tv_sec - start->tv_sec) * 1000) +
136  (now.tv_nsec - start->tv_nsec) / 1000000);
137  assert(elapsed >= 0);

142 bool acc_libgpiod_wait_for_interrupt(int pin, uint32_t timeout_ms)
143 {
144  assert(gpios[pin].direction == GPIO_DIR_INPUT_INTERRUPT);
145 
146  struct timespec start;
147  clock_gettime(CLOCK_MONOTONIC, &start);
148 
149  int pin_value = gpiod_line_get_value(gpios[pin].line);
150  assert(pin_value >= 0);
151 
152  unsigned int elapsed_ms = get_elapsed_ms(&start);
153 
154  while ((pin_value != PIN_HIGH) && (elapsed_ms < timeout_ms))
155  {
156  timeout_ms -= elapsed_ms;
157  time_t secs = timeout_ms / 1000;
158  long ns = (timeout_ms % 1000) * 1000000;
159  struct timespec ts = { secs, ns };
160  int res = gpiod_line_event_wait(gpios[pin].line, &ts);
161  if (res < 0)
162  {
163  perror("gpiod_line_event_wait failed");
164  assert(true);
165  }
166  else if (res == 0)
167  {
168  // Timeout
169  break;
170  }
171  else
172  {
173  // Wait complete, check which event we got.
174  struct gpiod_line_event event;
175  res = gpiod_line_event_read(gpios[pin].line, &event);
176  if (res != 0)
177  {
178  perror("gpiod_line_event_read failed");
179  assert(true);
180  }
181 
182  if (event.event_type != GPIOD_LINE_EVENT_RISING_EDGE)
183  {
184  // The enums were renumbered between v1.0 and v1.1, make sure gpiod.h matches
185  // the shared library version if this happens.
186  fprintf(stderr, "Unexpected event_type: %d (expected %d), library version mismatch?\n",
187  event.event_type, GPIOD_LINE_EVENT_RISING_EDGE);
188  assert(true);
189  }
190 
191  pin_value = gpiod_line_get_value(gpios[pin].line);
192  if (pin_value < 0)
193  {
194  perror("gpiod_line_get_value failed");
195  assert(true);
196  }
197  }
198 
199  elapsed_ms = get_elapsed_ms(&start);
200  }
201  return pin_value == PIN_HIGH;
202 }
