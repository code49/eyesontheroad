27 bool acc_socket_server_open(acc_socket_server_t *socket_server, int server_port, size_t buffer_size)
28 {
29  int s;
30  struct sockaddr_in addr;
31 
32  socket_server->buffer_size = buffer_size;
33  socket_server->buffer = malloc(buffer_size);
34 
35  if (socket_server->buffer == NULL)
36  {
37  fprintf(stderr, "ERROR: Memory allocation error\n");
38  return false;
39  }
40 
41  if ((s = socket(AF_INET, SOCK_STREAM, 0)) < 0)
42  {
43  fprintf(stderr, "ERROR: socket(AF_INET, SOCK_STREAM, 0): (%u) %s\n", errno, strerror(errno));
44  free(socket_server->buffer);
45  socket_server->buffer = NULL;
46  return false;
47  }
48 
49  if (setsockopt(s, SOL_SOCKET, SO_REUSEADDR, &(int){1 }, sizeof(int)) < 0)
50  {
51  fprintf(stderr, "ERROR: setsockopt(SO_REUSEADDR): (%u) %s\n", errno, strerror(errno));
52  }
53 
54  memset(&addr, 0, sizeof(addr));
55  addr.sin_family = AF_INET;
56  addr.sin_addr.s_addr = INADDR_ANY;
57  addr.sin_port = htons(server_port);
58 
59  if (bind(s, (struct sockaddr *)&addr, sizeof(addr)) < 0)
60  {
61  fprintf(stderr, "ERROR: bind(): (%u) %s\n", errno, strerror(errno));
62  close(s);
63  free(socket_server->buffer);
64  socket_server->buffer = NULL;
65  return false;
66  }
67 
68  if (listen(s, 10) < 0)
69  {
70  fprintf(stderr, "ERROR: listen(): (%u) %s\n", errno, strerror(errno));
71  close(s);
72  free(socket_server->buffer);
73  socket_server->buffer = NULL;
74  return false;
75  }
76 
77  socket_server->server_socket = s;
78  socket_server->client_socket = -1;
79 
80  return true;
81 }

96 bool acc_socket_server_wait_for_client(acc_socket_server_t *socket_server)
97 {
98  /* Blocking wait for accept */
99  socket_server->client_socket = accept(socket_server->server_socket, NULL, NULL);
100 
101  if (socket_server->client_socket < 0)
102  {
103  return false;
104  }
105 
106  if (setsockopt(socket_server->client_socket, SOL_SOCKET, SO_KEEPALIVE, &(int){1 }, sizeof(int)) < 0)
107  {
108  fprintf(stderr, "ERROR:setsockopt(SO_KEEPALIVE): (%u) %s\n", errno, strerror(errno));
109  }
110 
111  if (setsockopt(socket_server->client_socket, IPPROTO_TCP, TCP_NODELAY, &(int){1 }, sizeof(int)) < 0)
112  {
113  fprintf(stderr, "ERROR:setsockopt(TCP_NODELAY): (%u) %s\n", errno, strerror(errno));
114  }
115 
116  if (setsockopt(socket_server->client_socket, SOL_SOCKET, SO_SNDBUF, &(int){200000 }, sizeof(int)) < 0)
117  {
118  fprintf(stderr, "ERROR:setsockopt(SO_SNDBUF): (%u) %s\n", errno, strerror(errno));
119  }
120 
121  socket_server->poll_set[0].fd = socket_server->client_socket;
122  socket_server->poll_set[0].events = POLLIN;
123 
124  return true;
125 }

139 bool acc_socket_server_poll_events(acc_socket_server_t *socket_server, bool blocking, size_t timeout_us)
140 {
141  bool success = true;
142 
143  struct timespec poll_wait = { 0 };
144  struct timespec *poll_wait_ptr = &poll_wait;
145 
146  if (blocking)
147  {
148  poll_wait_ptr = NULL;
149  }
150  else
151  {
152  poll_wait.tv_sec = timeout_us / US_TICKS_PER_SECOND;
153  poll_wait.tv_nsec = (timeout_us % US_TICKS_PER_SECOND) * NS_PER_TICKS;
154  }
155 
156  if (socket_server->client_socket > 0)
157  {
158  /* Wait for poll_wait time or until a socket event occurs */
159  int nof_events = ppoll(socket_server->poll_set, 1, poll_wait_ptr, NULL);
160 
161  if (nof_events > 0)
162  {
163  short returned_event = socket_server->poll_set[0].revents;
164  if (returned_event & POLLERR)
165  {
166  /* Socket error, break */
167  success = false;
168  }
169  else if (returned_event & POLLIN)
170  {
171  int len = read(socket_server->client_socket, socket_server->buffer, socket_server->buffer_size);
172  if (len >= 1)
173  {
174  /* Put data from socket */
175  if (socket_server->input_data_func != NULL)
176  {
177  socket_server->input_data_func(socket_server->buffer, len);
178  }
179  }
180  else if (len == 0)
181  {
182  /* Client socket disconnected */
183  success = false;
184  }
185  else
186  {
187  /* Interrupted by signal */
188  success = false;
189  }
190  }
191  }
192  else
193  {
194  /* POLL TIMEOUT or POLL ERROR, just continue */
195  }
196  }
197  else
198  {
199  /* Client socket has been closed */
200  success = false;
201  }
202 
203  return success;
204 }
