134 int main(int argc, char *argv[])
135 {
136  static struct option long_options[] =
137  {
138  {"help", no_argument, 0, 'h'},
139  {"log-level", required_argument, 0, 'l'},
140  {"port", required_argument, 0, 'p'},
141  {NULL, 0, NULL, 0}
142  };
143 
144  int character_code;
145  int option_index = 0;
146 
147  acc_log_level_t log_level = ACC_LOG_LEVEL_INFO;
148  int tcp_ip_port = DEFAULT_TCP_IP_PORT;
149 
150  while ((character_code = getopt_long(argc, argv, "h?l:p:", long_options, &option_index)) != -1)
151  {
152  switch (character_code)
153  {
154  case 'h':
155  case '?':
156  {
157  print_usage(basename(argv[0]));
158  return EXIT_FAILURE;
159  }
160  case 'l':
161  {
162  if (strcmp(optarg, "debug") == 0)
163  {
164  log_level = ACC_LOG_LEVEL_DEBUG;
165  }
166  else if (strcmp(optarg, "verbose") == 0)
167  {
168  log_level = ACC_LOG_LEVEL_VERBOSE;
169  }
170  else if (strcmp(optarg, "info") == 0)
171  {
172  log_level = ACC_LOG_LEVEL_INFO;
173  }
174  else if (strcmp(optarg, "warning") == 0)
175  {
176  log_level = ACC_LOG_LEVEL_WARNING;
177  }
178  else if (strcmp(optarg, "error") == 0)
179  {
180  log_level = ACC_LOG_LEVEL_ERROR;
181  }
182  else
183  {
184  fprintf(stderr, "ERROR: Unknown log-level '%s'\n", optarg);
185  return EXIT_FAILURE;
186  }
187 
188  break;
189  }
190  case 'p':
191  {
192  int value = atoi(optarg);
193  if (value > 0)
194  {
195  tcp_ip_port = value;
196  printf("Overriding tcp/ip port (%d)\n", tcp_ip_port);
197  }
198  else
199  {
200  fprintf(stderr, "ERROR: Invalid tcp/ip port '%s'\n", optarg);
201  return EXIT_FAILURE;
202  }
203  }
204  default:
205  break;
206  }
207  }
208 
209  acc_exploration_server_register_all_services();
210 
211  if (!acc_exploration_server_init(command_buffer, sizeof(command_buffer), "linux", log_level))
212  {
213  return EXIT_FAILURE;
214  }
215 
216  /* setup signal handler */
217  struct sigaction sa = { 0 };
218 
219  sa.sa_handler = main_sig_handler;
220  if (sigaction(SIGINT, &sa, NULL) < 0)
221  {
222  fprintf(stderr, "ERROR: sigaction\n");
223  cleanup();
224  return EXIT_FAILURE;
225  }
226 
227  /* Ignore broken pipe shutdown, default behavior is to close application */
228  signal(SIGPIPE, SIG_IGN);
229 
230  printf("Starting server (port=%d)\n", tcp_ip_port);
231 
232  if (!acc_socket_server_open(&socket_server, tcp_ip_port, MAX_COMMAND_SIZE))
233  {
234  fprintf(stderr, "ERROR: Could not create socket server\n");
235  cleanup();
236  return EXIT_FAILURE;
237  }
238 
239  acc_socket_server_set_input_data_func(&socket_server, input_data_function);
240 
241  while (!do_shutdown())
242  {
243  printf("Waiting for new connections...\n");
244  fflush(stdout);
245 
246  /* Close client socket if there was any */
247  acc_socket_server_client_close(&socket_server);
248 
249  /* Stop streaming if there was any */
250  acc_exploration_server_stop_streaming();
251 
252  /* Blocking accept */
253  if (!acc_socket_server_wait_for_client(&socket_server))
254  {
255  /* No valid accept, continue and try again... */
256  continue;
257  }
258 
259  printf("Got new connection.\n");
260  printf("Listening for command...\n");
261 
262  while (!do_shutdown())
263  {
264  /* Default state is idle */
265  acc_exploration_server_state_t state = ACC_EXPLORATION_SERVER_WAITING;
266 
267  /* Default wait time is zero */
268  int32_t ticks_until_next = 0;
269  bool blocking_poll = false;
270 
271  bool success = acc_exploration_server_process(&server_if, &state, &ticks_until_next);
272 
273  if (!success)
274  {
275  fprintf(stderr, "ERROR: acc_exploration_server_process (%u) %s\n", errno, strerror(errno));
276  break;
277  }
278 
279  switch (state)
280  {
281  case ACC_EXPLORATION_SERVER_STOPPED:
282  set_shutdown(true);
283  /* Stop received, do not wait for more events (continue) */
284  continue;
285  case ACC_EXPLORATION_SERVER_WAITING:
286  /* Wait blocking until an event occurs */
287  blocking_poll = true;
288  break;
289  case ACC_EXPLORATION_SERVER_STREAMING:
290  /* Wait full amount of us (or until a socket event occurs) */
291  break;
292  }
293 
294  if (!acc_socket_server_poll_events(&socket_server, blocking_poll, ticks_until_next))
295  {
296  /* Socket disconnected or other error */
297  break;
298  }
299  }
300  }
301 
302  acc_socket_server_client_close(&socket_server);
303 
304  acc_socket_server_close(&socket_server);
305 
306  cleanup();
307  printf("Shutdown complete.\n");
308 
309  return EXIT_SUCCESS;
310 }
